jobs/host.sync.ts 
import { defineJob } from 'jsr:@hushkey/remq';
export const hostSync = defineJob('host.sync', async (ctx) => {
  await ctx.logger('Syncing ' + ctx.data.hostId);
});
main.ts 
import { Remq } from 'jsr:@hushkey/remq';
import { hostSync } from './jobs/host.sync.ts';
const remq = Remq.create({ db, streamdb, expose: 4000 });
remq.on(hostSync).start();
remq.emit('host.sync', { hostId: 'abc' });

Features

defineJob

Type-safe standalone job files, TData inference.

Retry + DLQ

Per-job retry, shouldRetry filter, auto DLQ routing.

Cron scheduler

repeat.pattern, dedup on restart.

WebSocket gateway

ctx.socket.update(), per-job or broadcast.

RemqAdmin SDK

CRUD, pipeline-optimised, 1 round trip for getJob.

Queue stats

Per-status counts, dashboard-ready.

App context

TApp generic, services/db/mailer typed on ctx.

Tokio runtime

True parallel I/O, not single event loop.

Self-cleaning

TTL, MAXLEN, maxLogsPerJob, no manual cleanup.
